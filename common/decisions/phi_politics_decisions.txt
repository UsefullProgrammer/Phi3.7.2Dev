decision_phi_smother_rebellion = {
	owned_planets_only = yes
	sound = event_air_raid_siren
	icon = decision_martial_law

	resources = {
		category = decisions
		cost = {
			unity = 500
		}
	}

	potential = {
		always = no
		has_orbital_bombardment = no
		NOT = { has_planet_flag = phi_rebellion }
	}
	
	allow = {
		any_owned_pop = {
			has_job = rebel
		}
		NOT = {
			owner = {
				any_owned_planet = {
					has_ground_combat = yes
				}
			}
		}
	}

	effect = {
		custom_tooltip = decision_phi_smother_rebellion_effects
		hidden_effect = {
			planet_event = { id = phi_politics.0 }
		}
	}

	ai_weight = {
		weight = 0
	}
}

decision_phi_planet_expropriation = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_luxuries

	resources = {
		category = decisions
		cost = {
			trigger = {
				owner = { is_gestalt = no }
				branch_office_owner = {
					NOT = { has_overlord = root }
				}
			}
			unity = 1000
		}
		cost = {
			trigger = {
				branch_office_owner = {
					has_overlord = root
				}
			}
			unity = 100
		}
		
		cost = {
			trigger = {
				owner = { is_gestalt = yes }
			}
			influence = 100
		}
	}

	potential = {
		has_branch_office = yes
		branch_office_owner = {
			OR = {
				NOT = { has_valid_civic = civic_criminal_heritage }
				has_overlord = root
			}
		}
		owner = { is_gestalt = no }
	}

	allow = {
		owner = {
			root = {
				OR = {
					branch_office_owner = {
						is_at_war_with = prevprev
					}
					branch_office_owner = {
						has_overlord = root
					}
				}
			}
		}
	}

	effect = {
		close_branch_office = yes
		add_modifier = { modifier = "phi_planet_expropriation" days = 1800 }
		set_timed_planet_flag = { flag = criminal_branch_closed_cooldown_planet@this days = 3600 }
	}

	ai_weight = {
		weight = 100
	}
}
