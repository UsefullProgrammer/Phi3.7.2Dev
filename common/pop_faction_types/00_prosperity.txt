##############################
# Peaceful Prosperity Faction
##############################

prosperity = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_pacifist

	unique = yes
	support_multiplier = 1

	is_potential = {
		years_passed > 1
		NOT = { has_ethic = ethic_gestalt_consciousness }
		NOT = { has_country_flag = suppress_vanilla_factions }
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	resources = {
		category = pop_factions
		produces = {
			influence = 2
			unity = 10
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		NOT = { has_trait = trait_nerve_stapled }
	}

	attraction = {
		base = 100

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.0
			OR = {
				leader_class = admiral
				leader_class = general
			}
		}

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}
	
	demand = {
		title = "TRADITIONAL_MERCANTILISM"
		unfulfilled_title = "NO_TRADITIONAL_MERCANTILISM"
		desc = "TRADITIONAL_MERCANTILISM_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				has_tradition = tr_mercantile_finish
			}
		}
	}

	demand = {
		title = "PROSPERITY_PHILOSOPHY"
		unfulfilled_title = "PROSPERITY_NO_PHILOSOPHY"
		desc = "PROSPERITY_PHILOSOPHY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = {
				any_relation = {
					has_communications = root.owner
					is_country_type = default
					NOT = { exists = overlord }
				}
			}
		}

		trigger = {
			owner = {
				has_policy_flag = no_wars
			}
		}
	}

	demand = {
		title = "PROSPERITY_PEACE"
		unfulfilled_title = "PROSPERITY_NO_PEACE"
		desc = "PROSPERITY_PEACE_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -5

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				is_at_war = no
			}
		}
	}

	demand = {
		title = "PROSPERITY_NOT_AGGRESSOR"
		unfulfilled_title = "PROSPERITY_AGGRESSOR"
		desc = "PROSPERITY_NOT_AGGRESSOR_DESC"

		unfulfilled_effect = -25
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 1
			}
		}

		trigger = {
			owner = {
				NOR = {
					has_country_flag = entered_war_as_attacker
					any_war = { is_war_participant = { who = prev side = attackers } }
				}
			}
		}
	}
	
	demand = {
		title = "NO_INDISCRIMINATE_BOMBING"
		unfulfilled_title = "INDISCRIMINATE_BOMBING"
		desc = "INDISCRIMINATE_BOMBING_DESC"

		unfulfilled_effect = -20
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 1
			}
		}

		trigger = {
			owner = {
				has_policy_flag = orbital_bombardment_selective
			}
		}
	}	

	demand = {
		title = "DEMOBILIZED_MILITARY"
		unfulfilled_title = "MOBILIZED_MILITARY"
		desc = "PEACE_DEMOBILIZED_MILITARY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				is_homicidal = no
				is_country_type = default
				OR = {
					has_policy_flag = diplo_stance_disarmament
					has_policy_flag = diplo_stance_minimal_army
				}
			}
		}

		trigger = {
			owner = {
				OR = {
					has_policy_flag = diplo_stance_disarmament
					has_policy_flag = diplo_stance_minimal_army
				}
			}
		}
	}

	demand = {
		title = "DEMOBILIZED_MILITARY"
		unfulfilled_title = "MOBILIZED_MILITARY"
		desc = "PEACE_MOBILIZED_MILITARY_DESC"

		fulfilled_effect = 0.001
		unfulfilled_effect = -20

		potential = {
			exists = owner
			owner = {
				is_homicidal = no
				is_country_type = default
				OR = {
					has_policy_flag = diplo_stance_conscription
					has_policy_flag = diplo_stance_full_scale_mobilization
				}
			}
		}

		trigger = {
			owner = {
				NOR = {
					has_policy_flag = diplo_stance_conscription
					has_policy_flag = diplo_stance_full_scale_mobilization
				}
			}
		}
	}

	demand = {
		title = "PEACE_5Y"
		unfulfilled_title = "NO_PEACE_5Y"
		desc = "PEACE_5Y"

		fulfilled_effect = 10
		unfulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 0
			}
		}

		trigger = {
			owner = {
				years_of_peace = {
					value > 4
				}
			}
		}
	}

	demand = {
		title = "ISOLATIONIST_PEACE2"
		unfulfilled_title = "ISOLATIONIST_NO_PEACE2"
		desc = "ISOLATIONIST_PEACE_DESC2"

		fulfilled_effect = 20
		unfulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 0
				years_of_peace = {
					value > 4
				}
			}
		}

		trigger = {
			owner = {
				years_of_peace = {
					value > 9
				}
			}
		}
	}
	
	demand = {
		title = "ISOLATIONIST_NON_AGGRESSION1"
		unfulfilled_title = "ISOLATIONIST_FEW_NON_AGGRESSION1"
		desc = "ISOLATIONIST_NON_AGGRESSION_DESC1"

		fulfilled_effect = 5
		unfulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 1
			}
		}

		trigger = {
			owner = {
				any_neighbor_country = {
					OR = {
						has_non_aggression_pact = root.owner
						has_association_status = root.owner
						is_in_federation_with = root.owner
					}
				}
			}
		}
	}

	on_create = {
		
	}

	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_pacifist
			}
			remove_modifier = promoted_pacifist
		}
		if = {
			limit = {
				has_modifier = suppressed_pacifist
			}
			remove_modifier = suppressed_pacifist
		}
	}

	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"

			cost = {
				influence = 500
			}

			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { is_subject_type = dominion }
					}
				}
			}

			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = { has_modifier = embraced_faction_timer }
					}
				}
				support > 0.20
				parameter:empire = {
					NOT = { has_ethic = "ethic_fanatic_pacifist" }
				}
			}

			effect = {
				add_modifier = { modifier = embraced_faction days = 3600 }
				parameter:empire = {
					shift_ethic = ethic_pacifist
					hidden_effect = {
						add_modifier = { modifier = embraced_faction_timer days = 3600 }
						every_pop_faction = {
							limit = { NOT = { is_same_value = root } }
							add_modifier = { modifier = embraced_another_faction days = 3600 }
						}
					}
				}
				hidden_effect = {
					save_event_target_as = TargetFaction
					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default
								has_communications = prev
								has_intel_level = {
									who = prev
									category = government
									level >= 1
								}
							}
							country_event = { id = factions.2000 }
						}
					}
				}
			}

			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.50
					owner = {
						has_ethic = ethic_pacifist
					}
				}
				modifier = {
					factor = 0
					owner = { has_valid_civic = civic_fanatic_purifiers }
				}
				modifier = {
					factor = 0
					NOT = {
						owner = {
							exists = ruler
							ruler = {
								leader_of_faction = prosperity
							}
						}
					}
				}
				modifier = {
					factor = 0
					owner = {
						count_pop_faction = {
							count < 4
						}
					}
				}
			}
		}
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_pacifist
						has_modifier = promoted_pacifist
					}
				}
			}

			effect = {
				parameter:empire = {
					add_modifier = { modifier = promoted_pacifist days = -1 }
				}
			}

			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = promoted_pacifist
				}
			}

			effect = {
				parameter:empire = {
					remove_modifier = promoted_pacifist
				}
			}

			ai_weight = {
				base = 0
			}
		}
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_pacifist
						has_modifier = promoted_pacifist
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = suppressed_pacifist days = -1 }
				}
			}

			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = suppressed_pacifist
				}
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_pacifist
				}
			}

			ai_weight = {
				base = 0
			}
		}
	}
}
