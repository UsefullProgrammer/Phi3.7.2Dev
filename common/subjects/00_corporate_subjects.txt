	joins_overlord_wars = no

	diplo_threat = 1.5

	intel_from_being_subject = 30
	intel_from_being_overlord = 100

	subject_tax_rate = {

		minerals = 0.05
		alloys = 0.05

		exotic_gases = 0.05
		rare_crystals = 0.05
		physics_research = 0.05
		society_research = 0.05
		engineering_research = 0.05

	potential = {
		is_country_type = default
		from = { has_authority = auth_corporate }
		has_been_declared_crisis = no
	}

	effect = {
		if = {
			limit = {
				from = {
					is_country_type_with_subjects = no

			set_subject_of = {
				who = none

			break = yes

			set_subject_of = {
				who = from
				subject_type = tributary
			}
			break = yes
		}
		if = {
			limit = {
				from = {
					has_valid_civic = civic_hive_devouring_swarm
				}
			}
			set_subject_of = {
				who = none
			}
			break = yes
		}
		if = {
			limit = {
				NOT = { is_same_species = from }

					has_valid_civic = civic_fanatic_purifiers
				}
			}
			set_subject_of = {
				who = none
			}
			break = yes
		}

			limit = {
				from = {
					is_country_type = awakened_fallen_empire
					has_ethic = ethic_fanatic_materialist
				}
			}
			set_subject_of = {
				who = from
				subject_type = satellite
			}
			break = yes
		}

			limit = {
				from = {
					is_country_type = awakened_fallen_empire

			}

				who = from
				subject_type = dominion
			}
			break = yes
		}
		if = {
			limit = {
				from = {
					is_country_type = awakened_fallen_empire
					has_ethic = ethic_fanatic_xenophile
				}
			}
			set_subject_of = {
				who = from
				subject_type = signatory
			}
			break = yes
		}
		if = {
			limit = {
				from = {
					is_country_type = awakened_fallen_empire
					has_ethic = ethic_fanatic_xenophobe
				}
			}
			set_subject_of = {
				who = from
				subject_type = thrall
			}
			break = yes
		}
		if = {
			limit = {
				from = { NOT = { has_authority = auth_corporate } }
			}
			set_subject_of = {
				who = from
				subject_type = vassal
			}
			break = yes
		}
	}
