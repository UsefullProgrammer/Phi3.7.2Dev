
FW_wg_open_border = {
	casus_belli = FW_cb_border
	hide_if_no_cb = yes
	set_defender_wargoal = FW_wg_open_border
	#defender_default = yes
	release_occupied_systems_on_status_quo = yes
	cede_claims = no
	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}
	surrender_acceptance = 25
	
	potential = {
		has_total_war_cb = no
	}
	on_accept = {
        custom_tooltip = z_fw_open_border_tooltip
        hidden_effect = {
            from = {
				set_closed_borders = {
				   who = root
				   status = no
				   forced = 20
				}
			}
        }
	}
		
	ai_weight = {
		weight = 30
		modifier = {
			factor = 1.5
			check_variable = { which = FW_var_cntr_DICE_1 value > 4 }
		}
		modifier = {
			factor = 1.5
			check_variable = { which = FW_var_cntr_DICE_2 value > 4 }
		}
		modifier = {
			factor = 1.5
			check_variable = { which = FW_var_cntr_DICE_3 value > 5 }
		}
	}
}

	#defender_default = yes
	release_occupied_systems_on_status_quo = yes
	cede_claims = no
	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}
	surrender_acceptance = -30
	
	potential = {
		has_total_war_cb = no
	}
	on_accept = {
		custom_tooltip = z_fw_open_border_long_tooltip

			end_rivalry = from

				set_closed_borders = {
				   who = root

			}
        }
	}
		
	ai_weight = {
		weight = 30
		modifier = {
			factor = 1.5

		}
		modifier = {
			factor = 1.5
			check_variable = { which = FW_var_cntr_DICE_2 value > 3 }
		}
		modifier = {
			factor = 1.5
			check_variable = { which = FW_var_cntr_DICE_3 value > 4 }
		}
	}
}
